    1. C++联机服务库的导入
      httplib库下载

    2. 联机的基本框架
      client.cpp

      #include "../thirdparty/httplib.h"


int main(int argc, char** argv)
{
    httplib::Client client("localhost:25565");

    httplib::Result result = client.Post("/hello");
    if (!result || result->status != 200)
    {
        std::cout << "error!" << result->status << std::endl;
        return -1;
    }

    std::cout << result->body << std::endl;

    system("pause");
    return 0;
}

      server.cpp

      #include "../thirdparty/httplib.h"

int main(int argc, char** argv)
{
    httplib::Server server;

    server.Post("/hello", [&](const httplib::Request& req, httplib::Response& res) {
        std::cout << "get client hello request!" << req.path << std::endl;

        res.set_content("Hello From Server!", "text/plain");
    });

    server.listen("localhost", 25565);
    return 0;
}

    客户端和服务端通过本地端口25565进行通信
